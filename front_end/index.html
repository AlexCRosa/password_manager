from pyscript import when import string import random @when("click", selector="#submit") def handle_click(event): password_length = int(Element("password_length").value) new_password = Element("new_password") selected_option = None if Element("strong").prop("checked") is not None and Element("strong").prop("checked"): selected_option = "strong" elif Element("weak").prop("checked") is not None and Element("weak").prop("checked"): selected_option = "weak" if not selected_option: console.log("Please select either a strong or weak password.") return password_characters = string.ascii_letters + string.digits if selected_option == "strong": password_characters += string.punctuation password = [] for _ in range(password_length): password.append(random.choice(password_characters)) new_password.write("".join(password))
